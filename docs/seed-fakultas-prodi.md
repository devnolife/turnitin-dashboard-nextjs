## Seed Script (TypeScript/Prisma)

```typescript
// Prisma Schema yang dibutuhkan:
// model Faculty {
//   id        String         @id @default(cuid())
//   name      String
//   code      String         @unique
//   programs  StudyProgram[]
// }
//
// model StudyProgram {
//   id        String   @id @default(cuid())
//   name      String
//   code      String   @unique
//   degree    String   @default("S1")
//   facultyId String
//   faculty   Faculty  @relation(fields: [facultyId], references: [id])
// }

interface ProdiData {
  kodeFakultas: string
  kodeProdi: string
  namaProdi: string
  kodeNim: string
  statusProdi: string
}

const FAKULTAS_MAP: Record<string, string> = {
  "01": "Fakultas Keguruan dan Ilmu Pendidikan",
  "02": "Fakultas Teknik",
  "03": "Fakultas Ekonomi dan Bisnis",
  "04": "Fakultas Pertanian",
  "05": "Fakultas Kedokteran dan Ilmu Kesehatan",
  "06": "Fakultas Agama Islam",
  "07": "Fakultas Ilmu Sosial dan Ilmu Politik",
  "08": "Fakultas Hukum",
  "09": "Program Pascasarjana",
}

function parseDegree(namaProdi: string): string {
  const lower = namaProdi.toLowerCase()
  if (lower.startsWith("s-1 ") || lower.startsWith("s1 ")) return "S1"
  if (lower.startsWith("s-2 ") || lower.startsWith("s2 ") || lower.startsWith("magister ")) return "S2"
  if (lower.startsWith("s-3 ") || lower.startsWith("s3 ")) return "S3"
  if (lower.startsWith("d-3 ") || lower.startsWith("d3 ")) return "D3"
  if (lower.startsWith("d-4 ") || lower.startsWith("d4 ")) return "D4"
  if (lower.startsWith("profesi ") || lower.includes("profesi")) return "Profesi"
  if (lower.startsWith("spesialis ")) return "Spesialis"
  return "S1"
}

async function fetchAllProdi(graphqlUrl: string): Promise<ProdiData[]> {
  const query = `
    query GetAllProdi {
      getAllProdi {
        kodeFakultas
        kodeProdi
        namaProdi
        kodeNim
        statusProdi
      }
    }
  `
  const response = await fetch(graphqlUrl, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ query }),
  })
  const result = await response.json()
  return result.data?.getAllProdi ?? []
}

async function seedProdi(prisma: any, graphqlUrl: string) {
  const prodiList = await fetchAllProdi(graphqlUrl)

  const groups = new Map<string, ProdiData[]>()
  for (const prodi of prodiList) {
    const arr = groups.get(prodi.kodeFakultas) ?? []
    arr.push(prodi)
    groups.set(prodi.kodeFakultas, arr)
  }

  for (const [kodeFakultas, prodiItems] of groups) {
    const namaFakultas = FAKULTAS_MAP[kodeFakultas] ?? `Fakultas ${kodeFakultas}`

    const faculty = await prisma.faculty.upsert({
      where: { code: kodeFakultas },
      update: { name: namaFakultas },
      create: { name: namaFakultas, code: kodeFakultas },
    })

    for (const prodi of prodiItems) {
      await prisma.studyProgram.upsert({
        where: { code: prodi.kodeProdi },
        update: {
          name: prodi.namaProdi,
          degree: parseDegree(prodi.namaProdi),
          facultyId: faculty.id,
        },
        create: {
          name: prodi.namaProdi,
          code: prodi.kodeProdi,
          degree: parseDegree(prodi.namaProdi),
          facultyId: faculty.id,
        },
      })
    }

    console.log(`✓ ${namaFakultas} (${prodiItems.length} prodi)`)
  }
}
```

## Query Pembayaran Mahasiswa

```graphql
query MahasiswaPembayaran($nim: String!, $jenisPembayaran: String!) {
  mahasiswaPembayaran(nim: $nim, jenisPembayaran: $jenisPembayaran) {
    nim
    nama
    periode
    jenisPembayaran
    waktuPembayaran
    jumlahPembayaran
    statusPembayaran
  }
}
```

**Jenis Pembayaran:**
- `"TURNITIN DIPLOMA S1"` — untuk mahasiswa S1
- `"TURNITIN DIPLOMA S2"` — untuk mahasiswa S2
- `"PERPUSTAKAAN"` — fallback jika turnitin tidak ditemukan

**Status Pembayaran:** `"1"` atau `"LUNAS"` = sudah bayar
